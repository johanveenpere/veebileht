def strToEmoji(string, emoji0, emoji1):
    out = ' '
    
    for l in string:
        out += (format(ord(l), 'b')) + ' '

def intToEmoji(integer, emoji0, emoji1):
# 0 ja 1 asendamine emojidega
    out = '{0:06b}'.format(integer).replace("1", emoji1).replace("0", emoji0)
    return out
